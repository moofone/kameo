# Local Tell

A local `tell` is fire-and-forget.

```rust
use kameo::{Actor, ActorRef};
use kameo::message::{Context, Message};

#[derive(Actor)]
struct Printer;

struct Print(pub String);

impl Message<Print> for Printer {
    type Reply = ();

    async fn handle(&mut self, msg: Print, _ctx: &mut Context<Self, Self::Reply>) {
        println!("{}", msg.0);
    }
}

# tokio_test::block_on(async {
let printer: ActorRef<Printer> = <Printer as Actor>::spawn(());
printer.tell(Print("hello".to_string())).send().await.unwrap();
# });
```
